        // Signature Pad Logic

        function setupSignaturePad(canvasId) {

            const canvas = document.getElementById(canvasId);

            const ctx = canvas.getContext('2d');

            let drawing = false;


            // Set canvas size for better resolution

            const scale = window.devicePixelRatio; // Get the device pixel ratio

            canvas.width = canvas.offsetWidth * scale;

            canvas.height = canvas.offsetHeight * scale;

            ctx.scale(scale, scale);


            ctx.lineWidth = 2;

            ctx.lineCap = 'round';

            ctx.strokeStyle = '#000';


            function getMousePos(e) {

                const rect = canvas.getBoundingClientRect();

                return {

                    x: (e.clientX - rect.left) / scale,

                    y: (e.clientY - rect.top) / scale

                };

            }


            function getTouchPos(e) {

                const rect = canvas.getBoundingClientRect();

                return {

                    x: (e.touches[0].clientX - rect.left) / scale,

                    y: (e.touches[0].clientY - rect.top) / scale

                };

            }


            canvas.addEventListener('mousedown', (e) => {

                drawing = true;

                const pos = getMousePos(e);

                ctx.beginPath();

                ctx.moveTo(pos.x, pos.y);

            });


            canvas.addEventListener('mousemove', (e) => {

                if (!drawing) return;

                const pos = getMousePos(e);

                ctx.lineTo(pos.x, pos.y);

                ctx.stroke();

            });


            canvas.addEventListener('mouseup', () => {

                drawing = false;

                ctx.closePath();

            });


            canvas.addEventListener('mouseleave', () => {

                drawing = false;

                ctx.closePath();

            });


            // Touch events for mobile

            canvas.addEventListener('touchstart', (e) => {

                e.preventDefault(); // Prevent scrolling

                drawing = true;

                const pos = getTouchPos(e);

                ctx.beginPath();

                ctx.moveTo(pos.x, pos.y);

            });


            canvas.addEventListener('touchmove', (e) => {

                e.preventDefault(); // Prevent scrolling

                if (!drawing) return;

                const pos = getTouchPos(e);

                ctx.lineTo(pos.x, pos.y);

                ctx.stroke();

            });


            canvas.addEventListener('touchend', () => {

                drawing = false;

                ctx.closePath();

            });

        }


        function clearSignature(canvasId) {

            const canvas = document.getElementById(canvasId);

            const ctx = canvas.getContext('2d');

            ctx.clearRect(0, 0, canvas.width, canvas.height);

        }


        // Initialize signature pads on window load

        window.onload = function() {

            setupSignaturePad('vendorSignaturePad');

            setupSignaturePad('amariSignaturePad');

        };

                const apiKey = ""; // Canvas will automatically provide the API key


                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

                const response = await fetch(apiUrl, {

                    method: 'POST',

                    body: JSON.stringify(payload)

                });


                const result = await response.json();


                if (result.candidates && result.candidates.length > 0 &&

                    result.candidates[0].content && result.candidates[0].content.parts &&

                    result.candidates[0].content.parts.length > 0) {

                    const generatedText = result.candidates[0].content.parts[0].text;

                    workDescriptionTextarea.value = generatedText;

                } else {

                    console.error('ບໍ່ສາມາດສ້າງຄຳອະທິບາຍໜ້າວຽກໄດ້:', result); // Could not generate work description

                    alert('ເກີດຂໍ້ຜິດພາດໃນການສ້າງຄຳອະທິບາຍໜ້າວຽກ. ກະລຸນາລອງໃໝ່ອີກຄັ້ງ.'); // An error occurred while generating the work description. Please try again.

                }

            } catch (error) {

                console.error('Error calling Gemini API:', error);

                alert('ເກີດຂໍ້ຜິດພາດໃນການເຊື່ອມຕໍ່ກັບ Gemini API. ກະລຸນາກວດສອບການເຊື່ອມຕໍ່ອິນເຕີເນັດຂອງທ່ານ.'); // An error occurred connecting to the Gemini API. Please check your internet connection.

            } finally {

                // Hide loading spinner and enable button

                loadingSpinner.classList.add('hidden');

                generateDescBtn.disabled = false;

                generateDescBtn.style.opacity = '1';

            }

        }

                useCORS: true, // Required for images from external sources like placehold.co

                logging: true,

                allowTaint: true // Allow tainted canvas for external images if needed

            });


            const imgData = canvas.toDataURL('image/png');

            const pdf = new jsPDF('p', 'mm', 'a4'); // 'p' for portrait, 'mm' for millimeters, 'a4' for A4 size


            const imgWidth = 210; // A4 width in mm

            const pageHeight = 297; // A4 height in mm

            const imgHeight = canvas.height * imgWidth / canvas.width; // Calculate image height based on A4 width and aspect ratio


            // If the captured image height is greater than A4 height, scale it down to fit one page

            let finalImgWidth = imgWidth;

            let finalImgHeight = imgHeight;


            if (finalImgHeight > pageHeight) {

                finalImgHeight = pageHeight;

                finalImgWidth = canvas.width * pageHeight / canvas.height; // Recalculate width to maintain aspect ratio

            }


            pdf.addImage(imgData, 'PNG', (imgWidth - finalImgWidth) / 2, (pageHeight - finalImgHeight) / 2, finalImgWidth, finalImgHeight);

            

            pdf.save('ແບບຟອມ Vendor.pdf');
